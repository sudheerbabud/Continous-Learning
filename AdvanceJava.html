JDBC - Architecture

JDBC Drivers is classified into 4 Types

  1. JDBC - ODBC Bridge Driver - thick driver --> obselete(Depricated)
  2. Native-API Driver - thick driver --> obselete
  3. Network Protocol Driver - thin driver
  4. Native Driver - thin driver
Steps to Connect Database
  1. Loading the Driver
  2. Connection establsih
  3. Statement
  4. ResultSet
  5. Close Connection
Statements in JDBC
  1. statement() - Used for static SQL queries.
  2. PrepatedStatement() - Used for precompiled SQL queries, with parameterized queries to avoid SQL injection.
  3. CallableStatement() - Used to execute stored procedures.
Executing Queries in JDBC
  1. executeUpdate() - used for insert, update, delete
  2. executeQuery() - used for select - returns ResultSet
  3. execute() - used for DDL - returns boolean

interface StudentInterface{
    public void insertStudent();
    public void updateStudent();
    public void selectStudentbyId(int x);
    public void selectStudent();
}
                            
class StudentData implements StudentInterface{
    DBConnection db = new DBConnection();
        public void insertStudent() {
            try{
                Connection con = db.getConnection();
                PreparedStatement ps = con.prepareStatement("insert into Student(sname,marks) values(?,?)");
                ps.setString(1,"Sudheer");
                ps.setInt(2, 98);
                int n = ps.executeUpdate();
                if(n>0) {
                    System.out.println(n + "(s) rows inserted");
                }else {
                    .out.println("Something went wrong!");
                }
                con.close();
            }catch(Exception e) {
                System.out.println(e);
            }
        }
                            
        public void updateStudent() {
            try{
                Connection con = db.getConnection();
                PreparedStatement ps = con.prepareStatement("update Student set marks=? where sno=?)");
                ps.setString(1,"Sudheer");
                ps.setInt(2, 120);
                int n = ps.executeUpdate();
                if(n>0) {
                    System.out.println(n + "(s) rows Updated");
                }else {
                    System.out.println("Something went wrong!");
                }
                con.close();
            }catch(Exception e) {
                System.out.println(e);
            }
        }
                            
        public void selectStudentbyId(int x) {
            try{
                Connection con = db.getConnection();
                ps = con.prepareStatement("select * from Student where sno=?");
                ps.setInt(1,x);
                ResultSet rs = ps.executeQuery();
                while(rs.next()) {
                    System.out.println(rs.getInt(1)+ ":"+rs.getString(2)+":"+rs.getInt(3));;
                }
                con.close();
            }catch(Exception e) {
                System.out.println(e);
            }
        }
            
        public void selectStudent() {
            try{
                Connection con = db.getConnection();
                ps = con.prepareStatement("select * from Student");
                ResultSet rs = ps.executeQuery();
                while(rs.next()) {
                    System.out.println(rs.getInt(1)+ ":"+rs.getString(2)+":"+rs.getInt(3));;
                }
                con.close();
            }catch(Exception e) {
                System.out.println(e);
            }
        }
